# MODULAR PLAYER CONTROLLER v3.0
# ================================
# A complete player controller with weapon switching, 
# stance system, full RPC support, and integration points.

extends CharacterBody3D
class_name ModularPlayer

# Signals for external integration
signal weapon_changed(weapon_type: int, weapon_name: String)
signal weapon_fired(weapon_type: int, damage: int, position: Vector3)
signal stance_changed(new_stance: String, old_stance: String)
signal health_changed(current_health: int, max_health: int)
signal player_died(killer_id: int, weapon_type: String)

# Weapon System Configuration
enum WeaponType { 
    GUN = 0, MELEE = 1, GRENADE = 2, NONE = 3
}

@export_category("WEAPON NODES")
@export var gun_node: Node3D = null
@export var melee_node: Node3D = null
@export var grenade_node: Node3D = null

# Combat Settings
@export var max_health: int = 300
@export var gun_damage: int = 20
@export var fire_rate: float = 0.2

# Movement Settings
@export var normal_speed: float = 2.0
@export var sprint_speed: float = 6.0
@export var jump_velocity: float = 4.5

# Network Sync Variables
@export var sync_velocity: Vector3 = Vector3.ZERO
@export var sync_is_dead: bool = false
@export var sync_stance: String = "normal"

# Initialize systems
func _ready():
    _setup_collision_system()
    _setup_raycast_system()
    _setup_input_actions()
    _update_weapon_visibility()
# MODULAR ANIMATION CONTROLLER v3.0
# =================================
# Advanced animation system for Modular Player Controller.

extends Node3D
class_name ModularAnimationController

@export_category("REQUIRED REFERENCES")
@export var animation_player: AnimationPlayer = null
@export var character_reference: ModularPlayer = null

# Animation Speeds
@export var idle_speed: float = 1.0
@export var walk_speed: float = 1.0
@export var run_speed: float = 1.0
@export var melee_speed: float = 1.5

# Main animation function
func animate(velocity: Vector3, stance: String, 
             is_firing: bool = false, is_aiming: bool = false, 
             look_target: Vector3 = Vector3.ZERO):
    
    # Determine animation based on state
    var animation_name = _determine_animation(
        character_reference.current_weapon,
        stance,
        is_firing,
        is_aiming,
        velocity.length() > 0.1,
        character_reference.is_running_mode,
        character_reference.is_on_floor()
    )
    
    # Play animation
    if animation_player.has_animation(animation_name):
        animation_player.play(animation_name)
        animation_player.speed_scale = _calculate_speed(animation_name, stance)
# MODULAR CAMERA CONTROLLER v3.0
# ===============================
# Complete camera control system for modular player.

extends Node3D
class_name ModularCameraController

@export_category("CAMERA SETTINGS")
@export var spring_arm: SpringArm3D = null
@export var mouse_sensitivity: float = 0.005
@export var invert_y_axis: bool = false
@export var vertical_limit_min: float = -PI/2
@export var vertical_limit_max: float = PI/2

# Handle mouse input
func _input(event: InputEvent):
    if event is InputEventMouseMotion:
        _handle_mouse_movement(event.relative)

func _handle_mouse_movement(relative: Vector2):
    if not spring_arm:
        return
    
    # Horizontal rotation
    rotate_y(-relative.x * mouse_sensitivity)
    
    # Vertical rotation
    var vertical_input = relative.y * mouse_sensitivity
    if invert_y_axis:
        vertical_input = -vertical_input
    
    spring_arm.rotate_x(-vertical_input)
    
    # Apply limits
    spring_arm.rotation.x = clamp(
        spring_arm.rotation.x, 
        vertical_limit_min, 
        vertical_limit_max
    )
