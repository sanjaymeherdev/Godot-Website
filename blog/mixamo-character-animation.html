📦 How to Install

  1. Copy the script — Use the copy button below to copy the full Python code.
  2. Open Blender → Go to Edit → Preferences → Add-ons.
  3. Click "Install…" and paste the code into a new text file? Actually easier: Save the script as mixamo_importer.py on your computer.
  4. Click "Install…" → select the .py file → "Install Add-on".
  5. Enable the add-on by checking the box next to "Animation: Mixamo Importer".
  6. Find the panel in 3D Viewport → Sidebar (N key) → Animation tab.
✅
Alternative: Run as script — You can also paste the code directly into Blender's Text Editor and click "Run Script". The panel will appear in the 3D Viewport sidebar.

🚀 How to Use

  1. Prepare your folder — Download multiple Mixamo animations (FBX format) and place them in a single folder.
  2. Open Blender and switch to the Animation workspace or any 3D Viewport.
  3. Press N to open the sidebar, go to the Animation tab.
  4. Click "Import Mixamo Folder" and select the folder containing your FBX files.
  5. Wait a few seconds — the script imports each FBX, extracts the action, names it after the file, enables Fake User, and deletes the imported armature.
  6. Open the Action Editor (Dopesheet → Action Editor) to see all your imported actions ready to be assigned to any rig!
💡
Pro Tip: When exporting from Mixamo, make sure to download FBX files with "Skin" and "Animation" only. The script works best with standard Mixamo exports.

import bpy
import os
from pathlib import Path

# ============================================
# MAIN IMPORT FUNCTION
# ============================================

def import_mixamo_animations(folder_path):
    """Import all Mixamo FBX files from a folder as actions with fake user enabled"""
    
    folder = Path(folder_path)
    
    if not folder.exists():
        print(f"❌ Folder does not exist: {folder}")
        return {'CANCELLED'}
    
    # Get all FBX files
    fbx_files = list(folder.glob("*.fbx"))
    
    if not fbx_files:
        print(f"❌ No FBX files found in: {folder}")
        return {'CANCELLED'}
    
    print(f"\n✓ Found {len(fbx_files)} FBX files")
    print("=" * 50)
    
    imported_count = 0
    
    for fbx_path in fbx_files:
        print(f"\nImporting: {fbx_path.name}")
        
        # Deselect all
        bpy.ops.object.select_all(action='DESELECT')
        
        # Import FBX
        bpy.ops.import_scene.fbx(
            filepath=str(fbx_path),
            use_anim=True,
            ignore_leaf_bones=True,
            force_connect_children=True,
            automatic_bone_orientation=False,
            use_prepost_rot=True
        )
        
        # Find imported armature
        imported_armatures = [obj for obj in bpy.context.selected_objects 
                            if obj.type == 'ARMATURE']
        
        if imported_armatures:
            armature = imported_armatures[0]
            
            # Get action name from filename
            action_name = fbx_path.stem
            
            # Handle duplicate names
            if action_name in bpy.data.actions:
                counter = 1
                new_name = f"{action_name}_{counter:03d}"
                while new_name in bpy.data.actions:
                    counter += 1
                    new_name = f"{action_name}_{counter:03d}"
                action_name = new_name
            
            # Rename and enable fake user
            if armature.animation_data and armature.animation_data.action:
                action = armature.animation_data.action
                action.name = action_name
                action.use_fake_user = True
                
                print(f"  ✓ Action created: '{action_name}'")
                print(f"  ✓ Fake user: ON")
                imported_count += 1
            
            # Store all imported objects for deletion
            objects_to_delete = bpy.context.selected_objects.copy()
            
            # Delete all imported objects (armature + meshes)
            for obj in objects_to_delete:
                bpy.data.objects.remove(obj, do_unlink=True)
    
    print("\n" + "=" * 50)
    print(f"✓ Import complete! {imported_count} actions imported")
    print(f"✓ All actions saved with fake user enabled")
    
    return {'FINISHED'}


# ============================================
# UI OPERATOR AND PANEL
# ============================================

class MIXAMO_OT_ImportAnimations(bpy.types.Operator):
    """Import all Mixamo FBX animations from selected folder"""
    bl_idname = "import_anim.mixamo_batch"
    bl_label = "Import Mixamo Animations"
    bl_options = {'REGISTER', 'UNDO'}
    
    directory: bpy.props.StringProperty(
        name="Folder Path",
        subtype='DIR_PATH'
    )
    
    def execute(self, context):
        if not self.directory:
            self.report({'ERROR'}, "No folder selected")
            return {'CANCELLED'}
        
        result = import_mixamo_animations(self.directory)
        
        if result == {'FINISHED'}:
            self.report({'INFO'}, f"Successfully imported animations")
        
        return result
    
    def invoke(self, context, event):
        context.window_manager.fileselect_add(self)
        return {'RUNNING_MODAL'}


class MIXAMO_PT_ImportPanel(bpy.types.Panel):
    """Panel for Mixamo animation import"""
    bl_label = "Mixamo Importer"
    bl_idname = "MIXAMO_PT_import_panel"
    bl_space_type = 'VIEW_3D'
    bl_region_type = 'UI'
    bl_category = "Animation"
    
    def draw(self, context):
        layout = self.layout
        
        layout.operator("import_anim.mixamo_batch", 
                       text="Import Mixamo Folder", 
                       icon='IMPORT')
        
        layout.separator()
        box = layout.box()
        box.label(text="Action Library Stats:", icon='INFO')
        
        total = len(bpy.data.actions)
        fake_user = sum(1 for a in bpy.data.actions if a.use_fake_user)
        
        row = box.row()
        row.label(text=f"Total Actions: {total}")
        row = box.row()
        row.label(text=f"With Fake User: {fake_user}")


# ============================================
# REGISTRATION
# ============================================

def register():
    bpy.utils.register_class(MIXAMO_OT_ImportAnimations)
    bpy.utils.register_class(MIXAMO_PT_ImportPanel)
    print("Mixamo Importer registered")


def unregister():
    bpy.utils.unregister_class(MIXAMO_PT_ImportPanel)
    bpy.utils.unregister_class(MIXAMO_OT_ImportAnimations)
    print("Mixamo Importer unregistered")


# ============================================
# RUN SCRIPT
# ============================================

if __name__ == "__main__":
    try:
        unregister()
    except:
        pass
    
    register()
    
    print("\n" + "=" * 50)
    print("Mixamo Importer Ready!")
    print("=" * 50)
    print("Find the panel in: 3D Viewport > Sidebar (N) > Animation tab")
    print("Click 'Import Mixamo Folder' to select your folder")
    print("=" * 50)

🛠️ Troubleshooting

⚠️
No actions imported? Make sure your FBX files actually contain animation data. Mixamo exports without animation if you don't select "Animation" as the download type.
🔧
Fake User not enabled? The script automatically enables Fake User on every imported action. Check the Action Editor — actions with a shield icon (🔒) have fake user and won't be deleted when unused.
📁
Can I import from subfolders? The script only looks at the root of the selected folder (not recursive). Place all FBX files directly in one folder.

🎯 Why This Script?

Manual FBX import in Blender is tedious: select file → import → rename action → enable fake user → delete armature. Repeat 50 times. This script reduces 50 manual imports to one click.

Perfect for game developers using Mixamo for character animations, indie devs building action libraries, or anyone tired of repetitive Blender tasks.
