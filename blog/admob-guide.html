AdManager.gd - Complete Test Implementation
extends Node
class_name AdManager

signal banner_loaded
signal interstitial_loaded
signal rewarded_loaded
signal reward_earned(amount, currency)
signal ad_failed_to_load(error_code)

var admob = null
var is_initialized = false

# TEST IDs - USE ONLY DURING DEVELOPMENT
const TEST_APP_ID = "ca-app-pub-3940256099942544~3347511713"
const TEST_BANNER_ID = "ca-app-pub-3940256099942544/6300978111"
const TEST_INTERSTITIAL_ID = "ca-app-pub-3940256099942544/1033173712"
const TEST_REWARDED_ID = "ca-app-pub-3940256099942544/5224354917"

func _ready():
    if Engine.has_singleton("AdMob"):
        admob = Engine.get_singleton("AdMob")
        print("AdMob plugin found")
        initialize_test_ads()
    else:
        print("AdMob plugin NOT found!")

func initialize_test_ads():
    # Always use TRUE for test mode during development
    var use_test_ads = true
    
    # Initialize AdMob
    admob.init(use_test_ads, TEST_APP_ID)
    
    # Set banner position (BOTTOM or TOP)
    admob.setBannerPosition(admob.BANNER.BOTTOM)
    
    # Load all ads (preload)
    admob.loadBanner(TEST_BANNER_ID, admob.BANNER)
    admob.loadInterstitial(TEST_INTERSTITIAL_ID)
    admob.loadRewardedVideo(TEST_REWARDED_ID)
    
    # Connect signals
    admob.connect("banner_loaded", Callable(self, "_on_banner_loaded"))
    admob.connect("interstitial_loaded", Callable(self, "_on_interstitial_loaded"))
    admob.connect("rewarded_video_loaded", Callable(self, "_on_rewarded_loaded"))
    admob.connect("rewarded", Callable(self, "_on_reward_earned"))
    admob.connect("banner_failed_to_load", Callable(self, "_on_ad_failed"))
    
    is_initialized = true
    print("AdMob initialized in TEST mode")

func show_banner():
    if is_initialized:
        admob.showBanner()

func hide_banner():
    if is_initialized:
        admob.hideBanner()

func show_interstitial():
    if is_initialized and admob.isInterstitialLoaded():
        admob.showInterstitial()
    else:
        print("Interstitial not loaded yet")

func show_rewarded_video():
    if is_initialized and admob.isRewardedVideoLoaded():
        admob.showRewardedVideo()
    else:
        print("Rewarded video not loaded yet")

# Signal handlers
func _on_banner_loaded():
    emit_signal("banner_loaded")
    print("Banner ad loaded")

func _on_interstitial_loaded():
    emit_signal("interstitial_loaded")
    print("Interstitial ad loaded")

func _on_rewarded_loaded():
    emit_signal("rewarded_loaded")
    print("Rewarded video loaded")

func _on_reward_earned(currency, amount):
    emit_signal("reward_earned", amount, currency)
    print("Reward earned: ", amount, " ", currency)
    # Give reward to player here

func _on_ad_failed(error_code):
    emit_signal("ad_failed_to_load", error_code)
    print("Ad failed to load: ", error_code)
GameScene.gd - Using Ads in Game
extends Node

@onready var ad_manager = $AdManager

func _ready():
    # Show banner when level starts
    ad_manager.show_banner()
    
    # Connect to reward signal
    ad_manager.connect("reward_earned", Callable(self, "_on_reward_earned"))

func _on_level_completed():
    # Show interstitial between levels
    ad_manager.show_interstitial()
    
    # Wait for ad to close, then proceed
    await get_tree().create_timer(1.0).timeout
    load_next_level()

func _on_player_died():
    # Offer rewarded ad for continue
    show_revive_option()

func show_revive_option():
    # Show UI with "Watch ad to revive" button
    # When clicked:
    ad_manager.show_rewarded_video()

func _on_reward_earned(amount, currency):
    if currency == "revive":
        revive_player()
    
    if currency == "coins":
        add_coins(amount)

func add_coins(amount):
    var current_coins = get_coins()
    set_coins(current_coins + amount)

AdMob Approval Process

Checklist Before Submission:

  • App is complete (not demo/placeholder)
  • All test IDs replaced with real IDs
  • Privacy policy accessible in-app
  • Ads don't cover gameplay
  • No auto-clicking or fraud
  • Content appropriate for age rating
  • App tested on multiple devices

Submission Steps:

  1. Build release APK with production AdMob IDs
  2. Test thoroughly on 3+ devices
  3. Go to AdMob dashboard → Apps → Your App
  4. Click "Complete setup" or "Get started"
  5. Upload your APK (they'll test it)
  6. Wait 2-5 days for review
Timeline:
  • AdMob Review: 2-5 business days
  • If rejected: Fix issues and resubmit (2-3 days)
  • Initial limited ad serving: First 30 days (normal)
  • Full ad serving: After establishing good traffic

Common Rejection Reasons & Solutions:

1. "Placeholder app" → Add more content/levels
2. "Poor user experience" → Reduce ad frequency
3. "Missing privacy policy" → Add accessible policy
4. "Non-functional ads" → Test with real IDs briefly
5. "Policy violation" → Review AdMob policies

Troubleshooting & Common Issues

Common Issues and Solutions
1. Ads not showing:
   - Check internet connection
   - Verify AdMob initialization
   - Check if test mode is enabled
   - Verify ad unit IDs are correct

2. "AdMob plugin not found":
   - Check plugin installation
   - Verify in Android export settings
   - Rebuild custom template

3. App crashes with ads:
   - Check Android permissions
   - Verify minSdk version (21+)
   - Check ProGuard rules

4. Test ads not showing "Test Ad" label:
   - Ensure admob.init(true) is called
   - Check test device configuration

5. Banner ads not positioning correctly:
   - Call setBannerPosition() after init
   - Check screen orientation settings
